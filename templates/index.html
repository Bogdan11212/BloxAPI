Getting Started

Start integrating with the Roblox platform in minutes

import requests # Get user information by ID response = requests.get('http://localhost:5000/api/users/1234567') user_data = response.json() print(user_data)

Search for users by username:

import requests # Search for users params = { 'keyword': 'builderman', 'limit': 5 } response = requests.get('http://localhost:5000/api/users/search', params=params) search_results = response.json() print(search_results)

Get detailed game information:

import requests # Get detailed game information response = requests.get('http://localhost:5000/api/games/1234567890/details') game_details = response.json() print(game_details)
// Get user information by ID fetch('http://localhost:5000/api/users/1234567') .then(response => response.json()) .then(data => { console.log(data); }) .catch(error => { console.error('Error:', error); });

RoliMon's Integration Example:
import requests # Get item value and details from RoliMon's response = requests.get('http://localhost:5000/api/external/rolimon/items/1365767') item_data = response.json() print(f"Item Name: {item_data['data']['name']}") print(f"Current Value: {item_data['data']['value']}") print(f"RAP: {item_data['data']['rap']}") print(f"Value Trend: {item_data['data']['trend']}")
Rblx.Trade Integration Example:
import requests # Calculate trade value using Rblx.Trade data = { "offer_items": [1365767, 31101391], # Offering Dominus Frigidus and Dominus Empyreus "request_items": [1028606, 617605] # Requesting Classic ROBLOX Fedora and Clockwork's Headphones } response = requests.post('http://localhost:5000/api/external/rblx-trade/trade-calculator', json=data) calculation = response.json() print(f"Offering Total Value: {calculation['data']['offer_value']}") print(f"Requesting Total Value: {calculation['data']['request_value']}") print(f"Value Difference: {calculation['data']['value_difference']}") print(f"Trade Recommendation: {calculation['data']['recommendation']}")
Market Trends from Roliverse:
import requests # Get market trends from Roliverse params = { "item_type": "limited", "time_period": "week" } response = requests.get('http://localhost:5000/api/external/roliverse/market/trends', params=params) trends = response.json() print("Top Rising Items This Week:") for item in trends['data']['rising_items'][:5]: print(f"{item['name']}: +{item['percentage_change']}% (Current Value: {item['value']})") print("\nTop Falling Items This Week:") for item in trends['data']['falling_items'][:5]: print(f"{item['name']}: {item['percentage_change']}% (Current Value: {item['value']})")
