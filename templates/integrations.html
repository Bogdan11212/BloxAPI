Integration Code Examples

Here are some ready-to-use code snippets showing how to integrate with our external service APIs.

import requests # Get item value and details from RoliMon's item_id = 1365767 # Dominus Frigidus response = requests.get(f'https://api.example.com/api/external/rolimon/items/{item_id}') item_data = response.json() print(f"Item Name: {item_data['data']['name']}") print(f"Current Value: {item_data['data']['value']}") print(f"RAP: {item_data['data']['rap']}") print(f"Value Trend: {item_data['data']['trend']}") # Get player RAP user_id = 1234567 response = requests.get(f'https://api.example.com/api/external/rolimon/players/{user_id}/rap') rap_data = response.json() print(f"Player RAP: {rap_data['data']['rap']}")

import requests # Calculate trade value using Rblx.Trade data = { "offer_items": [1365767, 31101391], # Offering Dominus Frigidus and Dominus Empyreus "request_items": [1028606, 617605] # Requesting Classic ROBLOX Fedora and Clockwork's Headphones } response = requests.post('https://api.example.com/api/external/rblx-trade/trade-calculator', json=data) calculation = response.json() print(f"Offering Total Value: {calculation['data']['offer_value']}") print(f"Requesting Total Value: {calculation['data']['request_value']}") print(f"Value Difference: {calculation['data']['value_difference']}") print(f"Trade Recommendation: {calculation['data']['recommendation']}") # Get player trade reputation user_id = 1234567 response = requests.get(f'https://api.example.com/api/external/rblx-trade/players/{user_id}/reputation') reputation = response.json() print(f"Reputation Score: {reputation['data']['score']}/10") print(f"Total Trades: {reputation['data']['total_trades']}") print(f"Completed Trades: {reputation['data']['completed_trades']}") print(f"Cancelled Trades: {reputation['data']['cancelled_trades']}")

import requests # Get market trends from Roliverse params = { "item_type": "limited", "time_period": "week" } response = requests.get('https://api.example.com/api/external/roliverse/market/trends', params=params) trends = response.json() print("Top Rising Items This Week:") for item in trends['data']['rising_items'][:5]: print(f"{item['name']}: +{item['percentage_change']}% (Current Value: {item['value']})") print("\nTop Falling Items This Week:") for item in trends['data']['falling_items'][:5]: print(f"{item['name']}: {item['percentage_change']}% (Current Value: {item['value']})") # Get demand index for an item item_id = 1365767 # Dominus Frigidus response = requests.get(f'https://api.example.com/api/external/roliverse/items/{item_id}/demand') demand = response.json() print(f"Demand Index: {demand['data']['demand_index']}/10") print(f"Demand Category: {demand['data']['demand_category']}") print(f"Trading Volume: {demand['data']['trading_volume']} per day")

import requests # Check if an item is projected item_id = 1365767 # Dominus Frigidus response = requests.get(f'https://api.example.com/api/external/rblx-values/items/{item_id}/projected') projected = response.json() print(f"Is Projected: {projected['data']['is_projected']}") if projected['data']['is_projected']: print(f"Projected Since: {projected['data']['projected_since']}") print(f"Real Value Estimate: {projected['data']['real_value_estimate']}") print(f"Inflation Percentage: {projected['data']['inflation_percentage']}%") # Get rising value items response = requests.get('https://api.example.com/api/external/rblx-values/market/rising?limit=10') rising = response.json() print("\nTop Rising Value Items:") for item in rising['data']['items']: print(f"{item['name']}: +{item['value_change']} in last {item['time_period']}")

// Get item value and details from RoliMon's const itemId = 1365767; // Dominus Frigidus fetch(`https://api.example.com/api/external/rolimon/items/${itemId}`) .then(response => response.json()) .then(data => { console.log(`Item Name: ${data.data.name}`); console.log(`Current Value: ${data.data.value}`); console.log(`RAP: ${data.data.rap}`); console.log(`Value Trend: ${data.data.trend}`); }) .catch(error => { console.error('Error:', error); }); // Get player RAP const userId = 1234567; fetch(`https://api.example.com/api/external/rolimon/players/${userId}/rap`) .then(response => response.json()) .then(data => { console.log(`Player RAP: ${data.data.rap}`); }) .catch(error => { console.error('Error:', error); });

// Calculate trade value using Rblx.Trade const tradeData = { offer_items: [1365767, 31101391], // Offering Dominus Frigidus and Dominus Empyreus request_items: [1028606, 617605] // Requesting Classic ROBLOX Fedora and Clockwork's Headphones }; fetch('https://api.example.com/api/external/rblx-trade/trade-calculator', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify(tradeData) }) .then(response => response.json()) .then(data => { console.log(`Offering Total Value: ${data.data.offer_value}`); console.log(`Requesting Total Value: ${data.data.request_value}`); console.log(`Value Difference: ${data.data.value_difference}`); console.log(`Trade Recommendation: ${data.data.recommendation}`); }) .catch(error => { console.error('Error:', error); }); // Get player trade reputation const userId = 1234567; fetch(`https://api.example.com/api/external/rblx-trade/players/${userId}/reputation`) .then(response => response.json()) .then(data => { console.log(`Reputation Score: ${data.data.score}/10`); console.log(`Total Trades: ${data.data.total_trades}`); console.log(`Completed Trades: ${data.data.completed_trades}`); console.log(`Cancelled Trades: ${data.data.cancelled_trades}`); }) .catch(error => { console.error('Error:', error); });

// Get market trends from Roliverse const params = new URLSearchParams({ item_type: 'limited', time_period: 'week' }); fetch(`https://api.example.com/api/external/roliverse/market/trends?${params}`) .then(response => response.json()) .then(data => { console.log("Top Rising Items This Week:"); data.data.rising_items.slice(0, 5).forEach(item => { console.log(`${item.name}: +${item.percentage_change}% (Current Value: ${item.value})`); }); console.log("\nTop Falling Items This Week:"); data.data.falling_items.slice(0, 5).forEach(item => { console.log(`${item.name}: ${item.percentage_change}% (Current Value: ${item.value})`); }); }) .catch(error => { console.error('Error:', error); }); // Get demand index for an item const itemId = 1365767; // Dominus Frigidus fetch(`https://api.example.com/api/external/roliverse/items/${itemId}/demand`) .then(response => response.json()) .then(data => { console.log(`Demand Index: ${data.data.demand_index}/10`); console.log(`Demand Category: ${data.data.demand_category}`); console.log(`Trading Volume: ${data.data.trading_volume} per day`); }) .catch(error => { console.error('Error:', error); });

// Check if an item is projected const itemId = 1365767; // Dominus Frigidus fetch(`https://api.example.com/api/external/rblx-values/items/${itemId}/projected`) .then(response => response.json()) .then(data => { console.log(`Is Projected: ${data.data.is_projected}`); if (data.data.is_projected) { console.log(`Projected Since: ${data.data.projected_since}`); console.log(`Real Value Estimate: ${data.data.real_value_estimate}`); console.log(`Inflation Percentage: ${data.data.inflation_percentage}%`); } }) .catch(error => { console.error('Error:', error); }); // Get rising value items fetch('https://api.example.com/api/external/rblx-values/market/rising?limit=10') .then(response => response.json()) .then(data => { console.log("\nTop Rising Value Items:"); data.data.items.forEach(item => { console.log(`${item.name}: +${item.value_change} in last ${item.time_period}`); }); }) .catch(error => { console.error('Error:', error); });
